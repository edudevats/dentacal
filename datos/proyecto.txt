class Cita(db.Model):

    estado = db.Column(db.String(20), default='pendiente') # pendiente, confirmada, no_asistencia

3. Lógica del Servidor y Validación (app.py)
Lo más importante aquí es la validación de disponibilidad: no puede haber dos citas al mismo tiempo en el mismo consultorio.

Python
from flask import Flask, render_template, request, jsonify
from models import db, Dentista, Paciente, Cita
from datetime import datetime, timedelta

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///consultorio.db'
db.init_app(app)

@app.route('/')
def index():
    dentistas = Dentista.query.all()
    return render_template('index.html', dentistas=dentistas)

@app.route('/api/crear_cita', methods=['POST'])
def crear_cita():
    data = request.json
    inicio = datetime.fromisoformat(data['inicio'])
    fin = datetime.fromisoformat(data['fin'])
    consultorio = data['consultorio']

    # VALIDACIÓN: ¿Está el consultorio ocupado?
    ocupado = Cita.query.filter(
        Cita.consultorio == consultorio,
        Cita.fecha_inicio < fin,
        Cita.fecha_fin > inicio
    ).first()

    if ocupado:
        return jsonify({"status": "error", "message": "El consultorio ya está ocupado a esa hora"}), 400

    nueva_cita = Cita(
        fecha_inicio=inicio,
        fecha_fin=fin,
        consultorio=consultorio,
        dentista_id=data['dentista_id'],
        paciente_id=data['paciente_id']
    )
    db.session.add(nueva_cita)
    db.session.commit()
    return jsonify({"status": "ok"})

@app.route('/api/eventos')
def obtener_eventos():
    citas = Cita.query.all()
    eventos = []
    for c in citas:
        eventos.append({
            "title": f"C{c.consultorio} - {c.paciente.nombre}",
            "start": c.fecha_inicio.isoformat(),
            "end": c.fecha_fin.isoformat(),
            "color": c.dentista.color, # Aquí aplicamos el color del doctor
            "extendedProps": {"estado": c.estado, "dentista": c.dentista.nombre}
        })
    return jsonify(eventos)

if __name__ == '__main__':
    with app.app_context():
        db.create_all()
    app.run(debug=True)
